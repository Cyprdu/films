// Récupérer le bouton et la vidéo var playButton = document.getElementById('playButton'); var video = document.getElementById('video'); var videoContainer = document.querySelector('.video-container'); // Fonction pour extraire un identifiant unique de l'URL de la vidéo (domaine + chemin) function getDomainFromUrl(url) { try { const urlObj = new URL(url); return urlObj.origin + urlObj.pathname; // Utiliser domaine et chemin comme clé unique } catch (e) { console.error("URL invalide :", url); return url; // Si l'URL est invalide, renvoyer la chaîne brute } } // Fonction pour appliquer la position sauvegardée de la vidéo function applySavedTime(video, key) { const savedTime = localStorage.getItem('videoTime_' + key); if (savedTime) { video.currentTime = parseFloat(savedTime); // Appliquer la position sauvegardée console.log(`Reprise de la vidéo à ${savedTime} secondes.`); } } // Enregistrer la position de lecture toutes les 5 secondes video.addEventListener('timeupdate', function () { const videoKey = getDomainFromUrl(video.src); // Identifier la vidéo par son URL localStorage.setItem('videoTime_' + videoKey, video.currentTime); // Sauvegarder la position de lecture }); // Reprendre la lecture à la dernière position sauvegardée au chargement de la page document.addEventListener('DOMContentLoaded', function () { const videoKey = getDomainFromUrl(video.src); // Identifier la vidéo par son URL applySavedTime(video, videoKey); // Appliquer la position sauvegardée }); // Fonction pour mettre la vidéo en plein écran function goFullScreen() { if (video.requestFullscreen) { video.requestFullscreen(); } else if (video.mozRequestFullScreen) { // Firefox video.mozRequestFullScreen(); } else if (video.webkitRequestFullscreen) { // Chrome, Safari et Opera video.webkitRequestFullscreen(); } else if (video.msRequestFullscreen) { // IE/Edge video.msRequestFullscreen(); } } // Lorsque le bouton Play est cliqué playButton.addEventListener('click', function() { // Ajouter l'animation du bouton playButton.classList.add('clicked'); // Cacher la vignette et le bouton Play videoContainer.style.backgroundImage = 'none'; playButton.style.display = 'none'; // Afficher et lire la vidéo video.style.display = 'block'; video.play(); // Passer la vidéo en plein écran après avoir cliqué sur Play goFullScreen(); }); // Vérifie si Hls.js est supporté par le navigateur if (Hls.isSupported()) { var hls = new Hls(); // Charger la vidéo const videoSrc = 'https://matoc22148.cloudmatoc.com/files/aa/cWvcY9VU7dvdKQYuIcgIlyQ6YfqYCXzPT51.m3u8'; // Remplacer par l'URL HLS de votre vidéo hls.loadSource(videoSrc); hls.attachMedia(video); hls.on(Hls.Events.MANIFEST_PARSED, function () { const videoKey = getDomainFromUrl(videoSrc); applySavedTime(video, videoKey); // Appliquer la position sauvegardée lorsque le manifeste est chargé }); } else if (video.canPlayType('application/vnd.apple.mpegurl')) { // Cas pour les vidéos HLS dans les navigateurs qui supportent nativement HLS const videoSrc = 'https://matoc22148.cloudmatoc.com/files/aa/cWvcY9VU7dvdKQYuIcgIlyQ6YfqYCXzPT51.m3u8'; // Remplacer par l'URL HLS video.src = videoSrc; const videoKey = getDomainFromUrl(video.src); applySavedTime(video, videoKey); // Appliquer la position sauvegardée directement } else { alert("Votre navigateur ne supporte pas HLS."); } // Rediriger si la vidéo est terminée video.addEventListener('ended', function() { window.location.href = 'https://iptv01.netlify.app/tv'; });
